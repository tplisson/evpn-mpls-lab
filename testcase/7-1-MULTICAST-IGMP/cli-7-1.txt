
# 1- DISABLE IGMP snooping for EVI 30 on all LEAF routers
################# LEAF
set cli timestamp
configure
deactivate routing-instances vlan30 protocols igmp-snooping
commit and-quit

# 2- Verify that multicast traffic in EVI 30 is forwarded from the connected source on L-DC1-1A to ALL EVI 30 interfaces (even where no interested receiver is registered)
################# LEAF
show configuration routing-instances vlan30
set cli timestamp
show route table vlan30 match-prefix 6:*
show evpn igmp-snooping proxy extensive
show igmp snooping evpn database instance vlan30 detail
show route advertising-protocol bgp 10.35.2.1 table vlan30 match-prefix 6:*
show interfaces ae0 | match rate
show interfaces ae1 | match rate
show interfaces ae10 | match rate
